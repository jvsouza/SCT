clear;
clc;
s = tf('s');

Gpz1(s), Gpz2(s), Gpz3(s) e Gpz4(s)

...

num = 26.25*(s + 4);
% Gpz1(s)
den = (s + 3.5)*(s + 5)*(s + 6);    % polo em -3.5
step(num/den);
hold on
% Gpz2(s)
den = (s + 4.01)*(s + 5)*(s + 6);   % polo em -4.01
step(num/den);
hold on
% Gpz3(s)
den = (s + 4.1)*(s + 5)*(s + 6);    % polo em -4.10
step(num/den);
hold on
% Gpz4(s)
num = (26.25);
den = (s + 5)*(s + 6);              % polo em -4.00
step(num/den);
